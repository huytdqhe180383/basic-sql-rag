The database schema is:

TABLE: customers
  customer_id         INTEGER PRIMARY KEY
  zip                 INTEGER NOT NULL REFERENCES geography(zip)
  signup_date         DATE NOT NULL
  gender              TEXT NOT NULL                -- male, female, other
  age_group           TEXT NOT NULL                -- e.g. 18-24, 25-34
  acquisition_channel TEXT NOT NULL                -- e.g. organic_search, social_media

TABLE: orders
  order_id       INTEGER PRIMARY KEY
  order_date     DATE NOT NULL
  customer_id    INTEGER NOT NULL REFERENCES customers(customer_id)
  zip            INTEGER NOT NULL REFERENCES geography(zip)
  order_status   TEXT NOT NULL                    -- e.g. delivered, shipped, cancelled
  payment_method TEXT NOT NULL                    -- e.g. credit_card, cod
  device_type    TEXT NOT NULL                    -- e.g. mobile, desktop, tablet
  order_source   TEXT NOT NULL                    -- e.g. app, web, paid_search

TABLE: order_items
  order_item_id   INTEGER PRIMARY KEY
  order_id        INTEGER NOT NULL REFERENCES orders(order_id)
  product_id      INTEGER NOT NULL REFERENCES products(product_id)
  quantity        INTEGER NOT NULL
  unit_price      NUMERIC(14,2) NOT NULL
  discount_amount NUMERIC(14,2) NOT NULL
  promo_id        TEXT                            -- may be NULL, 'NO_PROMO' means no promo
  promo_id_2      TEXT                            -- secondary promo, may be empty or NULL

TABLE: products
  product_id   INTEGER PRIMARY KEY
  product_name TEXT NOT NULL
  category     TEXT NOT NULL
  segment      TEXT NOT NULL
  size         TEXT NOT NULL
  color        TEXT NOT NULL
  price        NUMERIC(14,2) NOT NULL
  cogs         NUMERIC(14,2) NOT NULL

TABLE: geography
  zip      INTEGER PRIMARY KEY
  city     TEXT NOT NULL
  region   TEXT NOT NULL
  district TEXT NOT NULL

TABLE: sales
  Date    DATE NOT NULL
  Revenue NUMERIC(14,2) NOT NULL
  COGS    NUMERIC(14,2) NOT NULL

TABLE: inventory
  snapshot_date    DATE NOT NULL
  product_id       INTEGER NOT NULL REFERENCES products(product_id)
  stock_on_hand    INTEGER NOT NULL
  units_received   INTEGER NOT NULL
  units_sold       INTEGER NOT NULL
  stockout_days    INTEGER NOT NULL
  days_of_supply   NUMERIC(14,2) NOT NULL
  fill_rate        NUMERIC(5,4) NOT NULL
  stockout_flag    BOOLEAN NOT NULL
  overstock_flag   BOOLEAN NOT NULL
  reorder_flag     BOOLEAN NOT NULL
  sell_through_rate NUMERIC(5,4) NOT NULL
  year             INTEGER NOT NULL
  month            INTEGER NOT NULL

RELATIONSHIPS:
  customers.customer_id  →  orders.customer_id
  customers.zip          →  geography.zip
  orders.zip             →  geography.zip
  orders.order_id        →  order_items.order_id
  order_items.product_id →  products.product_id
  inventory.product_id   →  products.product_id

IMPORTANT RULES:
- All 7 tables are in the public schema (no schema prefix needed).
- Use standard PostgreSQL syntax. Return ONLY the SQL query, nothing else.
- Only SELECT statements. Never INSERT, UPDATE, DELETE, DROP, CREATE, or ALTER.
