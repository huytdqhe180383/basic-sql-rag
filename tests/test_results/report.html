Section: Tell me the 3 most profitable products (total revenue minus cogs) ever, but exclude items currently flagged as overstock
Attempts: 1
WITH latest_inventory AS (
    SELECT DISTINCT ON (product_id) product_id, overstock_flag
    FROM inventory
    ORDER BY product_id, snapshot_date DESC
)
SELECT 
    p.product_name,
    SUM((oi.unit_price * oi.quantity) - (p.cogs * oi.quantity) - oi.discount_amount) AS total_profit

JOIN products p ON oi.product_id = p.product_id
LEFT JOIN latest_inventory li ON p.product_id = li.product_id
WHERE COALESCE(li.overstock_flag, FALSE) = FALSE
GROUP BY p.product_name
ORDER BY total_profit DESC

Section: Find the district with the highest refund rate (returned orders / total orders). How much revenue did we lose to those returns?
Attempts: 1
WITH district_stats AS (
    SELECT 
        g.district,
        COUNT(CASE WHEN o.order_status = 'returned' THEN 1 END) AS returned_orders,
        COUNT(*) AS total_orders,
        SUM(CASE WHEN o.order_status = 'returned' THEN oi.quantity * oi.unit_price - oi.discount_amount ELSE 0 END) AS revenue_lost
    FROM orders o
    JOIN geography g ON o.zip = g.zip
    JOIN order_items oi ON o.order_id = oi.order_id
    GROUP BY g.district
),
refund_rates AS (
    SELECT 
        district,
        returned_orders::FLOAT / total_orders AS refund_rate,
        revenue_lost
    FROM district_stats
)
SELECT 
    district,
    refund_rate,
    revenue_lost
FROM refund_rates
ORDER BY refund_rate DESC
LIMIT 1

    SELECT MAX(snapshot_date) AS snapshot_date
    FROM inventory
),

    SELECT 
        p.category,

Q10: We are looking at people who signed up in 2018. Which source gives us the most loyal buyers (most orders per person), and what color of clothing do they like to buy most?
Section: We are looking at people who signed up in 2018. Which source gives us the most loyal buyers (most orders per person), and what color of clothing do they like to buy most?
Attempts: 2
WITH customers_2018 AS (
    SELECT 
        customer_id,
        acquisition_channel
    FROM customers
    WHERE signup_date >= '2018-01-01' AND signup_date <= '2018-12-31'
),
customer_orders AS (
    SELECT 
        c.customer_id,
        c.acquisition_channel,
        COUNT(o.order_id) AS total_orders
    FROM customers_2018 c
    JOIN orders o ON c.customer_id = o.customer_id
    GROUP BY c.customer_id, c.acquisition_channel
),
channel_avg_orders AS (
    SELECT 
        acquisition_channel,
        AVG(total_orders) AS avg_orders_per_person
    FROM customer_orders
    GROUP BY acquisition_channel
),
top_channel AS (
    SELECT acquisition_channel
    FROM channel_avg_orders
    ORDER BY avg_orders_per_person DESC
    LIMIT 1
),
top_channel_customers AS (
    SELECT co.customer_id
    FROM customer_orders co
    JOIN top_channel tc ON co.acquisition_channel = tc.acquisition_channel
),
customer_colors AS (
    SELECT 
        p.color,
        SUM(oi.quantity) AS total_quantity
    FROM top_channel_customers tc
    JOIN orders o ON tc.customer_id = o.customer_id
    JOIN order_items oi ON o.order_id = oi.order_id
    JOIN products p ON oi.product_id = p.product_id
    WHERE p.category IN ('streetwear', 'outdoor', 'casual', 'genz')
    GROUP BY p.color
)
SELECT 
    tc.acquisition_channel AS source,
    cc.color AS favorite_color
FROM top_channel tc
CROSS JOIN (
    SELECT color
    FROM customer_colors
    ORDER BY total_quantity DESC
    LIMIT 1
) cc
